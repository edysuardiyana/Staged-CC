import csv 

from collections import namedtuple 

import operator 

import featurescom 

 

THRESHOLD_VALUE = 1.6 

ARRAY_TUPLED = namedtuple('ARRAY_TUPLED', 'AXC AYC AZC GXC GYC GZC AVMC GVMC ANNOTCHEST' 

                             ' AXT AYT AZT GXT GYT GZT AVMT GVMT ANNOTTHIGH')

def read_data(micro_path): 

    source_file = micro_path 

 

    chest_data = [] 

    chest_data_list = [] 

 

    with open(source_file) as object_file: 

        for line in object_file: 

            raw_data = line.split() 

            chest_data = [float(i) for i in raw_data] 

            named_chest_data = ARRAY_TUPLED(*chest_data) 

            chest_data_list.append(named_chest_data) 

    return chest_data_list 

 

def active_check(freq_rate,features_path,chest_data): 

 

    source_features = features_path 

    active_window = freq_rate * 2 #active window 

 

    buffer_pre_window = freq_rate 

    buffer_post_win_1 = freq_rate * 9 #active state is in first segment 

    buffer_post_win_2 = freq_rate * 10 #active state is in second segment 

 

    active_array = [] 

    vector_magnitude = [] 

 

    counter_clock = 0 #clock for active window 

    counter_post_win = 0 #clock for collecting post-impact data 

    post_window_flag = False 

 

    #open file to write instance 

    out_file = open(features_path, "w") 

    csv_writer = csv.writer(out_file, delimiter='\t') 

 

    stop_flag = False 

    index = 0 

 

    while index < len(chest_data): 

        print index 

        raw_data = ARRAY_TUPLED(*chest_data[index]) 

        if post_window_flag == False: 

            if (len(active_array)) < buffer_pre_window: 

                active_array.append(raw_data) 

            else: 

                if counter_clock == (active_window-1): #python indexing 

                    #print "hi this is index for 2 secs window" 

                    #print index 

                    active_array.append(raw_data) 

                    vector_magnitude.append(raw_data.AVMC) #collecting vector magnitude data for active state check 

                    index_max, max_value = find_max(vector_magnitude) #find maximum value 

                    if max_value < THRESHOLD_VALUE: #no active state in the 2 seconds window 

                        #reset the active window 

                        del active_array[:] 

                        del vector_magnitude[:] 

                        counter_clock = 0 

                    else: 

                        #there is an active state here, call the post impact data collection 

                        max_raw = active_array[buffer_pre_window + index_max] 

                        micro_annotation = max_raw.ANNOTCHEST 

 

                        if micro_annotation == 2: 

                            print "fall detected" 

 

                        active_state_value = max_value 

                        post_window_flag = True 

                        counter_clock = 0 #reset the clock for active window 

                        del vector_magnitude[:] #reset the vector magnitude buffer 

                else: 

                    vector_magnitude.append(raw_data.AVMC) #collecting vector magnitude data for active state check 

                    active_array.append(raw_data) 

                    counter_clock = counter_clock + 1 

        else: 

            if index_max <= (buffer_pre_window-1): # the active state is in the first segment 

                if counter_post_win == (buffer_post_win_1): 

                    #do feature calculation 

 

                    test_print(active_array) 

                    instance = featurescom.featurescom(active_array,active_state_value,micro_annotation, freq_rate) 

                    #writing the instance (collection of features) to file 

                    csv_writer.writerow(instance) 

 

                    counter_post_win = 0 

                    del vector_magnitude[:] #clean the vector magnitude buffer 

                    del active_array[:] #clean the active array buffer 

                    active_array.append(raw_data) # fill the empty with the current raw data to keep the counter right 

                    post_window_flag = False #start collecting pre-impact sample 

                else: 

                    active_array.append(raw_data) 

                    counter_post_win = counter_post_win + 1 

            else: # the active state is in the second segment 

                if counter_post_win == (buffer_post_win_2): 

                    #discard the first 1-second window buffer 

                    test_print(active_array[freq_rate:len(active_array)]) 

                    #do feature calculation 

                    instance = featurescom.featurescom(active_array[freq_rate:len(active_array)],active_state_value,micro_annotation,freq_rate) 

 

                    #writing the instance (collection of features) to file 

                    csv_writer.writerow(instance) 

 

                    counter_post_win = 0 

                    del vector_magnitude[:] 

                    del active_array[:] #clean the buffer 

                    active_array.append(raw_data) # fill the empty with the current raw data to keep the counter right 

                    post_window_flag = False #start collecting pre-impact sample 

                else: 

                    active_array.append(raw_data) 

                    counter_post_win = counter_post_win + 1 

        index = index + 1 

 

    out_file.close() 

 

def find_max(vector_magnitude): 

    index , max_val = max(enumerate(vector_magnitude),key=operator.itemgetter(1)) 

    return index,max_val
